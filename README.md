# TKD Videos

Uses this repo to generate videos

https://github.com/spuder/gifr

## Usage

```
for filename in *.env; do
    echo $filename
    unset RESOLUTION
    unset FPS
    unset VIDEO_URL
    source $filename
    docker run \
      -v $(pwd)/input:/input \
      -v $(pwd)/output:/output \
      -e VIDEO_URL \
      -e SPLITS \
      -e INPUT_FILE \
      -e OUTPUT_FILE \
      -e RESOLUTION \
      -e FPS \
      gifr
    unset RESOLUTION
    unset FPS
    unset VIDEO_URL
done
```
